## 1. Linear Search

```python
def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

nums = [4, 2, 7, 1, 9, 3]
print(linear_search(nums, 7))   # 2
print(linear_search(nums, 10))  # -1
```

## 2. Binary Search

```python
def binary_search(arr, target):
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

sorted_nums = [1, 3, 5, 7, 9, 11]
print(binary_search(sorted_nums, 7))  # 3
```
