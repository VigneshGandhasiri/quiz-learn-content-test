## 1. Reverse In-Place

```python
def reverse_list(arr):
    left = 0
    right = len(arr) - 1
    while left < right:
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1

nums = [1, 2, 3, 4, 5]
reverse_list(nums)
print(nums)  # [5, 4, 3, 2, 1]
```

## 2. Palindrome Check

```python
def is_palindrome(s):
    left, right = 0, len(s) - 1
    while left < right:
        if s[left] != s[right]:
            return False
        left += 1
        right -= 1
    return True

print(is_palindrome("racecar"))  # True
```
